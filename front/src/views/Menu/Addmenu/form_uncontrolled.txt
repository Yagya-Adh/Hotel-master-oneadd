import React, { useRef, useState } from "react";

const AddMenu = () => {
  const nameRef = useRef();
  const imageRef = useRef();
  const priceRef = useRef();

  const handleSubmit = (e) => {
    e.preventDefault();
    const foodname = nameRef.current.value;
    const foodprice = priceRef.current.value;
    const foodimage = imageRef.current.value;

    alert("name "+ foodimage + ",  "+ foodprice+ ",  "+ foodname);
  };
  return (
    <div>
      <form onSubmit={handleSubmit}>
        <h1>Add menu Items</h1>
        <div>
          <label htmlFor="">Name</label>
          <input type="text" name="name" id="" ref={nameRef} />
        </div>

        <div>
          <label htmlFor="">Price</label>
          <input type="number" name="price" id="" ref={priceRef} />
        </div>

        <div>
          <label htmlFor="">Image</label>
          <input type="file" ref={imageRef} />
        </div>

        <div>
          <input type="submit" value="Add menu" />
        </div>
      </form>
    </div>
  );
};

export default AddMenu;
